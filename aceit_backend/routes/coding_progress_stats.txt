@router.get("/progress/stats")
async def get_progress_stats(user_id: str):
    """
    Get comprehensive progress statistics for a user.
    Returns category-wise completion, difficulty distribution, and recent activity.
    """
    from sqlalchemy import text
    from collections import defaultdict
    from datetime import datetime, timedelta
    
    if not user_id:
        raise HTTPException(status_code=400, detail="user_id is required")
    
    db = SessionLocal()
    try:
        # Get all problems
        all_problems = get_db_problems(db)
        total_problems = len(all_problems)
        
        # Get solved problems for this user
        solved_query = text("SELECT problem_id, solved_at FROM user_coding_progress WHERE user_id = :user_id AND status = 'solved'")
        solved_result = db.execute(solved_query, {"user_id": user_id})
        solved_data = solved_result.fetchall()
        
        solved_problem_ids = [row[0] for row in solved_data]
        total_solved = len(solved_problem_ids)
        
        # Calculate category stats
        category_stats = defaultdict(lambda: {"solved": 0, "total": 0})
        for problem in all_problems:
            if problem.tags:
                tags_list = problem.tags if isinstance(problem.tags, list) else [problem.tags]
                for tag in tags_list:
                    category_stats[tag]["total"] += 1
                    if problem.id in solved_problem_ids:
                        category_stats[tag]["solved"] += 1
        
        # Calculate difficulty stats
        difficulty_stats = {
            "Easy": {"solved": 0, "total": 0},
            "Medium": {"solved": 0, "total": 0},
            "Hard": {"solved": 0, "total": 0}
        }
        
        for problem in all_problems:
            if problem.difficulty in difficulty_stats:
                difficulty_stats[problem.difficulty]["total"] += 1
                if problem.id in solved_problem_ids:
                    difficulty_stats[problem.difficulty]["solved"] += 1
        
        # Calculate recent activity (last 30 days)
        activity_data = defaultdict(int)
        today = datetime.now().date()
        
        for row in solved_data:
            if row[1]:  # solved_at timestamp
                solved_date = row[1].date() if hasattr(row[1], 'date') else row[1]
                days_ago = (today - solved_date).days
                if days_ago < 30:
                    activity_data[solved_date.isoformat()] += 1
        
        # Calculate current streak
        current_streak = 0
        max_streak = 0
        streak = 0
        check_date = today
        
        # Check for current streak (must include today or yesterday)
        for i in range(365):  # Check last year
            date_str = check_date.isoformat()
            if date_str in activity_data:
                streak += 1
                max_streak = max(max_streak, streak)
                if i == 0 or i == 1:  # Today or yesterday
                    current_streak = streak
            else:
                if i > 1:  # Break only after checking today and yesterday
                    break
                streak = 0
            check_date = check_date - timedelta(days=1)
        
        # Get recent solved problems (last 10)
        recent_solved = []
        for row in sorted(solved_data, key=lambda x: x[1] if x[1] else datetime.min, reverse=True)[:10]:
            problem = next((p for p in all_problems if p.id == row[0]), None)
            if problem:
                recent_solved.append({
                    "problem_id": problem.id,
                    "title": problem.title,
                    "difficulty": problem.difficulty,
                    "tags": problem.tags if isinstance(problem.tags, list) else [problem.tags] if problem.tags else [],
                    "solved_at": row[1].isoformat() if row[1] else None
                })
        
        # Calculate problems solved today
        today_str = today.isoformat()
        solved_today = activity_data.get(today_str, 0)
        
        return {
            "total_solved": total_solved,
            "total_problems": total_problems,
            "completion_percentage": round((total_solved / total_problems * 100), 1) if total_problems > 0 else 0,
            "current_streak": current_streak,
            "max_streak": max_streak,
            "solved_today": solved_today,
            "category_stats": dict(category_stats),
            "difficulty_stats": difficulty_stats,
            "recent_activity": [{"date": date, "count": count} for date, count in sorted(activity_data.items())],
            "recent_solved": recent_solved
        }
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"Error fetching progress stats: {str(e)}")
    finally:
        db.close()
